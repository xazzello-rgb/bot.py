// ─── MarkdownV2 helpers ───────────────────────────────────────────────────────

/** Escape all MarkdownV2 special characters */
function e(text: string): string {
  return String(text).replace(/[_*[\]()~`>#+\-=|{}.!\\]/g, "\\$&");
}

/** Bold */
function b(text: string): string { return `*${e(text)}*`; }

/** Italic */
function i(text: string): string { return `_${e(text)}_`; }

/** Inline code */
function c(text: string): string { return `\`${e(text)}\``; }

/** Raw bold — text already escaped */
function rb(text: string): string { return `*${text}*`; }

let _openai: OpenAI | null = null;
function getOpenAI(): OpenAI {
  if (!_openai) {
    const apiKey = process.env["OPENAI_API_KEY"] ?? "";
    const baseURL = apiKey.startsWith("gsk_") ? "https://api.groq.com/openai/v1" : undefined;
    _openai = new OpenAI({ apiKey, ...(baseURL ? { baseURL } : {}) });
  }
  return _openai;
}
function getModel(): string {
  return (process.env["OPENAI_API_KEY"] ?? "").startsWith("gsk_")
    ? "llama-3.3-70b-versatile" : "gpt-4o-mini";
}

interface TResult {
  translated: string;
  level: string;
  definition: string;
  synonyms: string[];
  detectedLang: string; // ISO code auto-detected by AI
}

// Map common AI-returned language names → our lang codes
const LANG_NAME_TO_CODE: Record<string, string> = {
  uzbek: "uz", russian: "ru", english: "en", turkish: "tr",
  arabic: "ar", german: "de", french: "fr", chinese: "zh",
  korean: "ko", japanese: "ja", spanish: "es", italian: "it",
  portuguese: "pt", ukrainian: "uk", kazakh: "kk",
  "o'zbek": "uz", "o'zbekcha": "uz", rus: "ru", ingliz: "en",
};

function normalizeLangCode(raw: string): string {
  const key = raw.toLowerCase().trim();
  // Direct code match
  if (LANGUAGES[key]) return key;
  // Name match
  return LANG_NAME_TO_CODE[key] ?? "en";
}

async function translateFull(text: string, tgtCode: string, attempt = 0): Promise<TResult> {
  const tgtEn = LANGUAGES[tgtCode]?.en ?? tgtCode;
  const isShort = text.trim().split(/\s+/).length <= 4;

  const systemPrompt = isShort
    ? `You are an expert multilingual dictionary assistant. Your job is to accurately translate words and short phrases, detect their source language, assign a CEFR level, and provide a simple child-friendly definition and synonyms.

RULES:
- Detect the source language precisely (e.g. Uzbek, Russian, English, Turkish, Arabic, etc.)
- Translate ACCURATELY into ${tgtEn} — do NOT guess or invent meaning
- For Uzbek (uz): it uses Latin script (a,b,d,e,f,g,h,i,j,k,l,m,n,o,p,q,r,s,t,u,v,x,y,z,oʻ,gʻ,sh,ch,ng) or Cyrillic — handle both
- definition must be written in the SAME language as the input text (the source language)
- synonyms must all be in ${tgtEn}
- CEFR level is for the TRANSLATED word/phrase in ${tgtEn}
- Output ONLY valid JSON — no markdown, no explanation, no extra text`
    : `You are an expert professional translator. Accurately translate texts between languages.

RULES:
- Detect the source language precisely
- Translate the FULL text accurately into ${tgtEn} — preserve meaning, tone, and nuance
- definition: a brief 1-sentence summary of the text written in the source language
- synonyms: 3 alternative phrasings/expressions in ${tgtEn}
- CEFR level: complexity estimate of the source text
- Output ONLY valid JSON — no markdown, no explanation, no extra text`;

  const userPrompt = isShort
    ? `Translate this word/phrase into ${tgtEn}:
"${text}"

Respond with this exact JSON structure:
{"detectedLang":"<ISO 639-1 code, e.g. uz/ru/en/tr/ar/de/fr/zh/ko/ja/es/it/pt/uk/kk>","translated":"<accurate translation in ${tgtEn}>","level":"<CEFR A1/A2/B1/B2/C1/C2>","definition":"<simple 1-sentence explanation in the source language>","synonyms":["<synonym1 in ${tgtEn}>","<synonym2 in ${tgtEn}>","<synonym3 in ${tgtEn}>"]}`
    : `Translate the following text into ${tgtEn}:
"${text}"

Respond with this exact JSON structure:
{"detectedLang":"<ISO 639-1 code>","translated":"<full accurate translation in ${tgtEn}>","level":"<CEFR A1/A2/B1/B2/C1/C2>","definition":"<1-sentence summary in source language>","synonyms":["<alt phrasing 1 in ${tgtEn}>","<alt phrasing 2 in ${tgtEn}>","<alt phrasing 3 in ${tgtEn}>"]}`  ;

  const res = await getOpenAI().chat.completions.create({
    model: getModel(),
    messages: [
      { role: "system", content: systemPrompt },
      { role: "user", content: userPrompt },
    ],
    max_tokens: 700,
    temperature: 0,
  });

  const raw = (res.choices[0]?.message?.content ?? "").trim();
  const match = raw.match(/\{[\s\S]*\}/);
  if (!match) {
    if (attempt < 2) return translateFull(text, tgtCode, attempt + 1);
    throw new Error("AI returned unparseable response after retries");
  }
  const parsed = JSON.parse(match[0]) as TResult;
  parsed.synonyms = (parsed.synonyms ?? []).filter(Boolean).slice(0, 5);
  parsed.level = (parsed.level ?? "").toUpperCase().replace(/^(A1|A2|B1|B2|C1|C2).*/, "$1");
  if (!["A1","A2","B1","B2","C1","C2"].includes(parsed.level)) parsed.level = "B1";
  parsed.detectedLang = normalizeLangCode(parsed.detectedLang ?? "");
  return parsed;
}

function formatResult(r: TResult, tgtCode: string, original: string): string {
  const src = LANGUAGES[r.detectedLang] ?? { flag: "🌐", local: r.detectedLang };
  const tgt = LANGUAGES[tgtCode] ?? { flag: "🌐", local: tgtCode };
  const levelLine = LEVEL_LABELS[r.level] ?? `📊 ${e(r.level)}`;
  const syns = r.synonyms.length
    ? r.synonyms.map(s => c(s)).join("  •  ")
    : e("—");

  return [
    `${e(src.flag)} ${e(src.local)} ➜ ${e(tgt.flag)} ${e(tgt.local)}`,
    "",
    `📝 ${b(original)}`,
    `➡️ ${b(r.translated)}`,
    "",
    levelLine,
    "",
    `💡 ${rb("Ta'rif:")} ${e(r.definition)}`,
    "",
    `🔄 ${rb("Sinonimlar:")} ${syns}`,
  ].join("\n");
}

// ─── Rate limiter ─────────────────────────────────────────────────────────────

const lastReq = new Map<number, number>();
function isRateLimited(uid: number): boolean {
  const now = Date.now();
  const last = lastReq.get(uid) ?? 0;
  if (now - last < RATE_MS) return true;
  lastReq.set(uid, now);
  return false;
}

async function upsertUser(from: {
  id: number; username?: string; first_name?: string; last_name?: string;
}) {
  const today = new Date().toISOString().slice(0, 10);
  const existing = await db.select().from(botUsers).where(eq(botUsers.telegramId, from.id)).limit(1);

  if (existing.length === 0) {
    await db.insert(botUsers).values({
      telegramId: from.id,
      username: from.username ?? null,
      firstName: from.first_name ?? null,
      lastName: from.last_name ?? null,
      sourceLang: "uz",
      targetLang: "en",
      lastStreakDate: today,
      streakDays: 1,
    });
    return;
  }

  const u = existing[0]!;
  const lastDate = u.lastStreakDate;
  const yesterday = new Date(Date.now() - 86400_000).toISOString().slice(0, 10);
  let streakDays = u.streakDays;
  let lastStreakDate = u.lastStreakDate;

  if (lastDate !== today) {
    if (lastDate === yesterday) { streakDays += 1; }
    else { streakDays = 1; }
    lastStreakDate = today;
  }

  await db.update(botUsers)
    .set({
      lastActiveAt: new Date(),
      username: from.username ?? null,
      firstName: from.first_name ?? null,
      lastStreakDate,
      streakDays,
    })
    .where(eq(botUsers.telegramId, from.id));
}

async function isAdmin(uid: number): Promise<boolean> {
  const env = (process.env["BOT_ADMIN_IDS"] ?? "").split(",").map(s => s.trim()).filter(Boolean);
  if (env.includes(String(uid))) return true;
  const rows = await db.select().from(botAdmins).where(eq(botAdmins.telegramId, uid)).limit(1);
  return rows.length > 0;
}

async function isBlocked(uid: number): Promise<boolean> {
  const rows = await db.select({ isBlocked: botUsers.isBlocked }).from(botUsers)
    .where(eq(botUsers.telegramId, uid)).limit(1);
  return rows[0]?.isBlocked ?? false;
}

async function getChannels() {
  return db.select().from(botChannels).orderBy(botChannels.addedAt);
}

async function checkSubs(bot: Telegraf, uid: number) {
  const channels = await getChannels();
  if (!channels.length) return { ok: true, missing: [] as { name: string; link: string }[] };
  const missing: { name: string; link: string }[] = [];
  for (const ch of channels) {
    try {
      const m = await bot.telegram.getChatMember(ch.channelId, uid);
      if (!["member", "administrator", "creator"].includes(m.status))
        missing.push({ name: ch.channelName, link: ch.channelLink });
    } catch {
      missing.push({ name: ch.channelName, link: ch.channelLink });
    }
  }
  return { ok: missing.length === 0, missing };
}

async function addHistory(uid: number, r: TResult, original: string, srcLang: string, tgtLang: string) {
  // Insert
  await db.insert(botTranslationHistory).values({
    telegramId: uid, original, translated: r.translated,
    sourceLang: srcLang, targetLang: tgtLang, level: r.level,
  });
  // Trim to HISTORY_LIMIT
  const ids = await db.select({ id: botTranslationHistory.id })
    .from(botTranslationHistory)
    .where(eq(botTranslationHistory.telegramId, uid))
    .orderBy(desc(botTranslationHistory.translatedAt));
  if (ids.length > HISTORY_LIMIT) {
    const oldIds = ids.slice(HISTORY_LIMIT).map(r => r.id);
    for (const id of oldIds) {
      await db.delete(botTranslationHistory).where(eq(botTranslationHistory.id, id));
    }
  }
  // Increment counter
  await db.update(botUsers)
    .set({ totalTranslations: sql`${botUsers.totalTranslations} + 1` })
    .where(eq(botUsers.telegramId, uid));
}

  const rows: ReturnType<typeof Markup.button.callback>[][] = [];
  for (let i = 0; i < POPULAR_LANGS.length; i += 3) {

function libPageKb(page: number, total: number) {
  const totalPages = Math.ceil(total / LIBRARY_PAGE);
  const btns: ReturnType<typeof Markup.button.callback>[] = [];
  if (page > 0) btns.push(Markup.button.callback("◀️", `lib_${page - 1}`));
  btns.push(Markup.button.callback(`${page + 1}/${totalPages}`, "lib_noop"));
  if ((page + 1) * LIBRARY_PAGE < total) btns.push(Markup.button.callback("▶️", `lib_${page + 1}`));
  const rows: ReturnType<typeof Markup.button.callback>[][] = [btns];
  rows.push([Markup.button.callback("🗑 Barchasini o'chirish", "clear_library")]);
  return Markup.inlineKeyboard(rows);
}

// ─── Library display ──────────────────────────────────────────────────────────

async function buildLibraryPage(uid: number, page: number) {
  const all = await db.select().from(botSavedWords)
    .where(eq(botSavedWords.telegramId, uid))
    .orderBy(desc(botSavedWords.savedAt));

  if (all.length === 0) return null;

  const pageItems = all.slice(page * LIBRARY_PAGE, (page + 1) * LIBRARY_PAGE);
  const lines = pageItems.map((w, i) => {
    const n = page * LIBRARY_PAGE + i + 1;
    const src = LANGUAGES[w.sourceLang];
    const tgt = LANGUAGES[w.targetLang];
    const lv = w.level ? ` \\(${e(w.level)}\\)` : "";
    const flags = src && tgt ? ` ${e(src.flag)}➜${e(tgt.flag)}` : "";
    return `${e(String(n))}\\. ${b(w.original)} ➜ ${b(w.translated)}${lv}${flags}`;
  });

  return {
    text: `📚 *Kutubxona* \\(${e(String(all.length))} ta so'z\\)\n\n${lines.join("\n")}`,
    kb: libPageKb(page, all.length),
  };
}

export function createBot(): Telegraf {
  const token = process.env["TELEGRAM_BOT_TOKEN"];
  if (!token) throw new Error("TELEGRAM_BOT_TOKEN is not set");

  const bot = new Telegraf(token);

  // ── Register Telegram command list ──────────────────────────────────────────

  void bot.telegram.setMyCommands([
    { command: "start",    description: "Botni boshlash" },
    { command: "help",     description: "Yordam va qo'llanma" },
    { command: "settings", description: "Tillarni sozlash" },
    { command: "library",  description: "Saqlangan so'zlar" },
    { command: "history",  description: "So'nggi tarjimalar" },
    { command: "mystats",  description: "Mening statistikam" },
    { command: "word",     description: "Kutubxonadan tasodifiy so'z (quiz)" },
    { command: "contact",  description: "Adminga murojat" },
    { command: "cancel",   description: "Amalni bekor qilish" },
  ]);

  // ── Guard middleware ─────────────────────────────────────────────────────────

  bot.use(async (ctx, next) => {
    const uid = ctx.from?.id;
    if (!uid) return next();

    // Always upsert user (updates streak, last active)
    if (ctx.from) {
      await upsertUser(ctx.from).catch(err => logger.warn({ err }, "upsertUser failed"));
    }

    if (await isBlocked(uid).catch(() => false)) {
      return ctx.reply("❌ Siz botdan bloklandingiz\\. Admin bilan bog'laning\\.", { parse_mode: "MarkdownV2" });
    }

    // Allow /start through without sub check
    const text = ctx.message && "text" in ctx.message ? ctx.message.text : "";
    if (text === "/start") return next();

    // Admins skip subscription check
    if (await isAdmin(uid).catch(() => false)) return next();

    const { ok, missing } = await checkSubs(bot, uid).catch(() => ({ ok: true, missing: [] }));
    if (!ok) {
      return ctx.reply(
        "📢 *Botdan foydalanish uchun quyidagi kanallarga obuna bo'ling:*",
        { parse_mode: "MarkdownV2", ...subKb(missing) }
      );
    }

    return next();
  });

  // ═══════════════════════════════════════════════════════════════════════════
  // COMMANDS
  // ═══════════════════════════════════════════════════════════════════════════

  // ── /start ──────────────────────────────────────────────────────────────────

  bot.command("start", async (ctx) => {
    const admin = await isAdmin(ctx.from!.id).catch(() => false);
    const tgtCode = await getUserTargetLang(ctx.from!.id).catch(() => "en");
    const tgt = LANGUAGES[tgtCode]!;
    const name = e(ctx.from!.first_name ?? "Do'st");

    await ctx.reply(
      `🌐 *Tarjimonation* ga xush kelibsiz, ${name}\\!\n\n` +
      `🔍 Manba til: *avtomatik aniqlanadi*\n` +
      `🎯 Tarjima tili: ${e(tgt.flag)} *${e(tgt.local)}*\n\n` +
      `Istalgan tilda so'z yoki gap yuboring — bot o'zi tilni aniqlab, tarjima qiladi\\!`,
      { parse_mode: "MarkdownV2", ...(admin ? adminKb() : mainKb()) }
    );
  });

  // ── /help ───────────────────────────────────────────────────────────────────

  const HELP_TEXT =
    `📖 *Tarjimonation — To'liq Qo'llanma*\n\n` +

    `━━━━━━━━━━━━━━━━━\n` +
    `🔤 *TARJIMA QILISH*\n` +
    `━━━━━━━━━━━━━━━━━\n` +
    `Istalgan tilda so'z yoki gap yuboring — bot tilni o'zi aniqlab tarjima qiladi\\.\n\n` +

  // ── /word (quiz) ─────────────────────────────────────────────────────────────

  const quizAnswers = new Map<number, string>(); // userId -> correct answer

  bot.command("word", async (ctx) => {
    const uid = ctx.from!.id;
    const words = await db.select().from(botSavedWords)
      .where(eq(botSavedWords.telegramId, uid));

    if (words.length < 1) {
      return ctx.reply(
        `📚 Kutubxonangiz bo'sh\\.\n\nAvval so'z saqlang, so'ngra /word buyrug'ini ishlating\\.`,
        { parse_mode: "MarkdownV2" }
      );
    }

    const pick = words[Math.floor(Math.random() * words.length)]!;
    quizAnswers.set(uid, pick.translated.toLowerCase());

    // Build wrong choices from other saved words
    const others = words.filter(w => w.id !== pick.id).sort(() => Math.random() - 0.5).slice(0, 3);
    const choices = [...others.map(w => w.translated), pick.translated]
      .sort(() => Math.random() - 0.5);

    const src = LANGUAGES[pick.sourceLang];
    const tgt = LANGUAGES[pick.targetLang];
    const flags = src && tgt ? `${e(src.flag)}➜${e(tgt.flag)}` : "";

    const btns = choices.map(ch => [Markup.button.callback(ch, `quiz_${ch}`)]);

    return ctx.reply(
      `🧠 *Quiz \\— so'zni toping\\!* ${flags}\n\n` +
      `❓ *"${e(pick.original)}"* — bu qaysi so'z?`,
      { parse_mode: "MarkdownV2", ...Markup.inlineKeyboard(btns) }
    );
  });

  bot.action(/^quiz_(.+)$/, async (ctx) => {

    const text = ctx.message.text.trim();
    const uid = ctx.from!.id;

    // Skip slash commands (handled above)
    if (text.startsWith("/")) return;

    const state = getAS(uid);

    // ── Admin broadcast flow ───────────────────────────────────────────────

    if (state.type === "broadcast") {
      setAS(uid, { type: "idle" });
      const users = await db.select({ telegramId: botUsers.telegramId })
        .from(botUsers).where(eq(botUsers.isBlocked, false));
      let sent = 0, failed = 0;
      const loading = await ctx.reply("📢 Yuborilmoqda…");
      for (const u of users) {
        try {
          await bot.telegram.sendMessage(u.telegramId, text);
          sent++;
        } catch { failed++; }
        if ((sent + failed) % 20 === 0) await new Promise(r => setTimeout(r, 1000));
      }
      await bot.telegram.deleteMessage(ctx.chat.id, loading.message_id).catch(() => {});
      return ctx.reply(
        `📢 *Broadcast yakunlandi\\!*\n✅ ${e(String(sent))} ta yuborildi\n❌ ${e(String(failed))} ta yuborilmadi`,
        { parse_mode: "MarkdownV2", ...adminKb() }
      );
    }

    // ── Channel add flow ───────────────────────────────────────────────────

    if (state.type === "ch_id") {
      setAS(uid, { type: "ch_name", channelId: text });
      return ctx.reply(`✅ ID: \`${e(text)}\`\n\n*2\\-qadam:* Kanal nomini yuboring:`, { parse_mode: "MarkdownV2" });
    }
    if (state.type === "ch_name") {
      setAS(uid, { type: "ch_link", channelId: state.channelId, channelName: text });
      return ctx.reply(`✅ Nom: *${e(text)}*\n\n*3\\-qadam:* Kanal linkini yuboring \\(https://t\\.me/\\.\\.\\.\\):`, { parse_mode: "MarkdownV2" });
    }
    if (state.type === "ch_link") {
      await db.insert(botChannels)
        .values({ channelId: state.channelId, channelName: state.channelName, channelLink: text })
        .onConflictDoNothing();
      setAS(uid, { type: "idle" });
      return ctx.reply(
        `✅ *Kanal qo'shildi\\!*\n📡 ${e(state.channelName)}\n🔗 ${e(text)}`,
        { parse_mode: "MarkdownV2", ...adminKb() }
      );
    }

    // ── Input validation ───────────────────────────────────────────────────

    if (!text) return;

    if (text.length > MAX_INPUT) {
      return ctx.reply(
        `⚠️ Matn juda uzun \\(${e(String(text.length))} ta belgi\\)\\.\n` +
        `Maksimal: *${e(String(MAX_INPUT))}* ta belgi\\.`,
        { parse_mode: "MarkdownV2", reply_parameters: { message_id: ctx.message.message_id } }
      );
    }

    // ── Rate limit ─────────────────────────────────────────────────────────

    if (isRateLimited(uid)) {
      return ctx.reply(
        "⏳ Biroz kuting… Ketma\\-ket yuborishni sekinlashtiring\\.",
        { parse_mode: "MarkdownV2", reply_parameters: { message_id: ctx.message.message_id } }
      );
    }

    // ── Translation ────────────────────────────────────────────────────────

    const targetLang = await getUserTargetLang(uid).catch(() => "en");
    await ctx.sendChatAction("typing");

    try {
      const result = await translateFull(text, targetLang);
      const detectedLang = result.detectedLang;

      // Store for save button & save to history (non-blocking)
      pendingSave.set(uid, {
        original: text, translated: result.translated,
        sourceLang: detectedLang, targetLang,
        level: result.level, definition: result.definition, synonyms: result.synonyms,
      });
      addHistory(uid, result, text, detectedLang, targetLang).catch(err =>
        logger.warn({ err }, "addHistory failed")
      );

      const formatted = formatResult(result, targetLang, text);
      return ctx.reply(formatted, {
        parse_mode: "MarkdownV2",
        reply_parameters: { message_id: ctx.message.message_id },
        ...saveKb(false),
      });
    } catch (err) {
      logger.error({ err }, "Translation failed");
      return ctx.reply(
        "❌ Tarjima qilishda xatolik yuz berdi\\. Iltimos, keyinroq urinib ko'ring\\.",
        { parse_mode: "MarkdownV2", reply_parameters: { message_id: ctx.message.message_id } }
      );
    }
  });

  // ── Global error handler ──────────────────────────────────────────────────

  bot.catch((err, ctx) => {
    logger.error({ err, update: ctx.update }, "Unhandled bot error");
    const uid = ctx.from?.id;
    if (uid) {
      ctx.reply("⚠️ Xatolik yuz berdi\\. Iltimos qayta urinib ko'ring\\.", { parse_mode: "MarkdownV2" }).catch(() => {});
    }
  });

  return bot;
}
